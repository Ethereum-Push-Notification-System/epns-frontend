```typescript
import { ChatViewComponent } from "@pushprotocol/uiweb";

const ChatViewComponentTest = () => {
  // you can add the amount of msgs that you want to fetch when the user first sees the component
  // by default the message fetch limit is 10.
  // by default image, file and gif option is there,
  // if you don't want any of them then just pass it as false inside the component
  // for example below commented code won't have the option for file, gif and emoji:
  /// <ChatViewComponent chatId='89f7b382b9b1496dca2b9a94ef87c80dd102db05dcb40b165fabfab28770fd55' emoji={false} gif={false} file={false} limit={10}/>
  return (
    <div>
      <ChatViewComponent
        chatId="89f7b382b9b1496dca2b9a94ef87c80dd102db05dcb40b165fabfab28770fd55"
        limit={10}
      />
    </div>
  );
};

export default ChatViewComponentTest;
```

---

| Param        | Type     | Default |
| ------------ | -------- | ------- |
| _`chatId`_   | string   | -       |
| messageInput | boolean  | true    |
| chatViewList | boolean  | true    |
| chatProfile  | boolean  | true    |
| limit        | number   | 10      |
| emoji        | boolean  | true    |
| gif          | boolean  | true    |
| file         | boolean  | true    |
| isConnected  | boolean  | true    |
| onGetTokenClick | function | void    |
| chatFilterList | Array | - |
| autoConnect | boolean | false |


> Note: Parameters _`in this style`_ are mandatory.
